Unidad 6: Definición de esquemas y vocabulario

2.1. XML asociado a DTD interna

DTD definido dentro del propio XML


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE marcadores [
	<!ELEMENT marcadores (pagina)*>
	<!ELEMENT pagina (nombre, descripcion, url)>
	<!ELEMENT nombre (#PCDATA)>
	<!ELEMENT descripcion (#PCDATA)>
	<!ELEMENT url (#PCDATA)>
]>
<marcadores>
	<pagina>
		<nombre>Programar fácil</nombre>
		<descripcion>Aprende a programar con arduino</descripcion>
		<url>https://programarfacil.com/</url>
	</pagina>
	<pagina>
		<nombre>Wikipedia</nombre>
		<descripcion>La enciclopedia libre.</descripcion>
		<url>http://www.wikipedia.org/</url>
	</pagina> 
	<pagina>
		<nombre>W3C</nombre>
		<descripcion>World Wide Web Consortium.</descripcion>
		<url>http://www.w3.org/</url>
	</pagina>
</marcadores>

2.2. XML asociado a DTD externa

DTD definido en archivo externo al XML

marcadores.dtd


<!ELEMENT marcadores (pagina)*>
<!ELEMENT pagina (nombre, descripcion, url)>
<!ELEMENT nombre (#PCDATA)>
<!ELEMENT descripcion (#PCDATA)>
<!ELEMENT url (#PCDATA)>

3.1.1.2. DTD: Elemento con subelementos

Elemento XML ciudad que contiene a los subelementos nombre y pais


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE ciudad [
	<!ELEMENT ciudad (nombre, pais)>
	<!ELEMENT nombre (#PCDATA)>
	<!ELEMENT pais (#PCDATA)>
]>
<ciudad>
	<nombre>Roma</nombre>
	<pais>Italia</pais>
</ciudad>

3.1.2.1. Cardinalidad de un elemento. Operador +

Ejemplo del operador +


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE personas [
	<!ELEMENT personas (nombre+)>
	<!ELEMENT nombre (#PCDATA)>
]>
<personas>
	<nombre>Ana</nombre>
	<nombre>Iker</nombre>
	<nombre>Elsa</nombre>
</personas>

3.1.2.2. Cardinalidad de un elemento. Operador *

Ejemplo del operador *


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE receta_de_cocina [
	<!ELEMENT receta_de_cocina (nombre, ingrediente*)>
	<!ELEMENT nombre (#PCDATA)>
	<!ELEMENT ingrediente (#PCDATA)>
]>
<receta_de_cocina>
	<nombre>Tortilla de patatas</nombre>
	<ingrediente>Huevo</ingrediente>
	<ingrediente>Patata</ingrediente>
	<ingrediente>Aceite</ingrediente>
	<ingrediente>Sal</ingrediente>
</receta_de_cocina>

3.2.1.1. Atributos. Valor por defecto

Declaración de un atributo con un valor por defecto


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE deportistas [
<!ELEMENT deportistas (futbolista | pilotof1 | tenista)*>
<!ELEMENT futbolista (#PCDATA)>
<!ELEMENT pilotof1 (#PCDATA)>
	<!ATTLIST pilotof1 pais CDATA "España">
<!ELEMENT tenista (#PCDATA)>
]>
<deportistas>
 <pilotof1 pais="Alemania">Sebastian Vettel</pilotof1>
 <pilotof1>Fernando Alonso</pilotof1>
 <tenista>Rafael Nadal</tenista>
</deportistas>

3.2.1.2. Atributos. Valores obligatorios

Declaración de un atributo obligatorio


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE deportistas [
<!ELEMENT deportistas (futbolista | pilotof1 | tenista)*>
<!ELEMENT futbolista (#PCDATA)>
<!ELEMENT pilotof1 (#PCDATA)>
<!ATTLIST pilotof1 pais CDATA "España">
<!ATTLIST pilotof1 escuderia CDATA #REQUIRED>
<!ELEMENT tenista (#PCDATA)>
]>
<deportistas>
 <pilotof1 escuderia="ferrari" pais="Alemania">Vettel</pilotof1>
 <tenista>Rafael Nadal</tenista>
</deportistas>

3.2.1.3. Atributos. Valores opcionales

Declaración de un atributo opcional


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE deportistas [
<!ELEMENT deportistas (futbolista | pilotof1 | tenista)*>
<!ELEMENT futbolista (#PCDATA)>
<!ELEMENT pilotof1 (#PCDATA)>
	<!ATTLIST pilotof1 pais CDATA #IMPLIED>
<!ELEMENT tenista (#PCDATA)>
]>
<deportistas>
 <pilotof1 pais="Alemania">Sebastian Vettel</pilotof1>
 <pilotof1>Fernando Alonso</pilotof1>
 <tenista>Rafael Nadal</tenista>
</deportistas>

3.2.1.4. Atributos. Valores fijos

Declaración de un atributo con valor fijo


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE deportistas [
<!ELEMENT deportistas (futbolista | pilotof1 | tenista)*>
<!ELEMENT futbolista (#PCDATA)>
<!ELEMENT pilotof1 (#PCDATA)>
	<!ATTLIST pilotof1 pais CDATA #IMPLIED>
	<!ATTLIST pilotof1 activo CDATA #FIXED "si">
<!ELEMENT tenista (#PCDATA)>
]>
<deportistas>
 <pilotof1 activo="si">Sebastian Vettel</pilotof1>
 <pilotof1>Fernando Alonso</pilotof1>
 <tenista>Rafael Nadal</tenista>
</deportistas>

3.2.2.1. Tipos de atributos. CDATA

Atributos de tipo CDATA: texto


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE peliculas [
<!ELEMENT peliculas (pelicula)*>
<!ELEMENT pelicula (#PCDATA)>
	<!ATTLIST pelicula pais CDATA #REQUIRED>
]>
<peliculas>
 <pelicula pais="EEUU">El padrino</pelicula>
 <pelicula pais="Brasil">Ciudad de Dios</pelicula>
</peliculas>

3.2.2.2. Tipos de atributos. Enumerados

Los atributos solo pueden tomar un valor procedente de una lista o enumerado


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE peliculas [
<!ELEMENT peliculas (pelicula)*>
<!ELEMENT pelicula (#PCDATA)>
	<!ATTLIST pelicula pais (ESP | EEUU | ITA | BRA) "ESP">
]>
<peliculas>
 <pelicula pais="EEUU">El padrino</pelicula>
 <pelicula pais="BRA">Ciudad de Dios</pelicula>
</peliculas>

3.2.2.3. Tipos de atributos. ID

Sólo permiten un valor único para cada elemento


<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE peliculas [
<!ELEMENT peliculas (pelicula)*>
<!ELEMENT pelicula (#PCDATA)>
	<!ATTLIST pelicula codigo ID #REQUIRED>
]>
<peliculas>
 <pelicula codigo="peli-00001">El padrino</pelicula>
 <pelicula codigo="peli-00002">Ciudad de Dios</pelicula>
</peliculas>
