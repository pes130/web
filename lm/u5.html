Unidad 5: Lenguajes para el almacenamiento y transmisión de la información

6.1. Ejemplo espacios de nombres

Documento con empleados y departamentos por empleado

Sin espacios de nombres


<?xml version="1.0" encoding="UTF-8"?>
<departamentos> 
  <departamento id="dpto#1">
    <nombre>Finanzas</nombre>
    <empleados>
      <empleado id="emp#1">
        <nombre>Juan</nombre>
        <apellidos>García Martínez</apellidos>
      </empleado>
      <empleado id="emp#2">
        <nombre>Pedro</nombre>
        <apellidos>Gutierrez Rovira</apellidos>
      </empleado>
    </empleados>
  </departamento>
  <departamento id="dpto#2">
    <nombre>Finanzas</nombre>
    <empleados>
      <empleado id="emp#3">
        <nombre>Juan</nombre>
        <apellidos>García Martínez</apellidos>
      </empleado>
      <empleado id="emp#4">
        <nombre>Pedro</nombre>
        <apellidos>Gutierrez Rovira</apellidos>
      </empleado>
    </empleados>
  </departamento>
</departamentos>

Con espacios de nombres


<?xml version="1.0" encoding="UTF-8"?>
<dept:departamentos xmlns:dept="empresa:departamentos" xmlns:emp="empresa:empleados">
  <dept:departamento dept:id="dpto#1">
    <dept:nombre>Finanzas</dept:nombre>
    <emp:empleados>
      <emp:empleado emp:id="emp#1">
        <emp:nombre>Juan</emp:nombre>
        <emp:apellidos>García Martínez</emp:apellidos>
      </emp:empleado>
      <emp:empleado id="emp#2">
        <emp:nombre>Pedro</emp:nombre>
        <emp:apellidos>Gutierrez Rovira</emp:apellidos>
      </emp:empleado>
    </emp:empleados>
  </dept:departamento>
  <dept:departamento dept:id="dpto#2">
    <dept:nombre>Finanzas</dept:nombre>
    <emp:empleados>
      <emp:empleado id="emp#3">
        <emp:nombre>Juan</emp:nombre>
        <emp:apellidos>García Martínez</emp:apellidos>
      </emp:empleado>
      <emp:empleado id="emp#4">
        <emp:nombre>Pedro</emp:nombre>
        <emp:apellidos>Gutierrez Rovira</emp:apellidos>
      </emp:empleado>
    </emp:empleados>
  </dept:departamento>
</dept:departamentos>

6.2. Espacio de nombres por defecto

Todas las etiquetas definidas bajo la definición del espacio de nombres, pertenecen a dicho espacio, y no necesitan prefijo

	
<!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN"
    "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd">
<html xmlns="http://www.w3.org/1999/xhtml" lang="es" xml:lang="es">
  <head>
    <title>Título de prueba</title>
    <meta http-equiv="Content-Type" content="text/html; charset=UTF8"/>
    <meta name="author" content="asir"/>
  </head>
  <body>
    <h1>título</h1>
    <p>Párrafo</p>
  </body>
</html>
